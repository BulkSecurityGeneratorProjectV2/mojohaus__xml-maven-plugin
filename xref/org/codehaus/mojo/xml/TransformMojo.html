86      private void setFeature( TransformerFactory pTransformerFactory, String name, Boolean value )
87          throws MojoExecutionException
88      {
89          // Try to use the method setFeature, which isn't available until JAXP 1.3
90          Method m;
91          try
92          {
93              m = pTransformerFactory.getClass().getMethod( "setFeature", new Class[]{ String.class, boolean.class } );
94          }
95          catch ( NoSuchMethodException e )
96          {
97              m = null;
98          }
99          if ( m == null )
100         {
101             // Not available, try to use setAttribute
102             pTransformerFactory.setAttribute( name, value );
103         }
104         else
105         {
106             try
107             {
108                 m.invoke( pTransformerFactory, new Object[]{ name, value } );
109             }
110             catch ( IllegalAccessException e )
111             {
112                 throw new MojoExecutionException( e.getMessage(), e );
113             }
114             catch ( InvocationTargetException e )
115             {
116                 Throwable t = e.getTargetException();
117                 throw new MojoExecutionException( t.getMessage(), t );
118             }
119         }

122     private Templates getTemplate( Resolver pResolver, Source stylesheet, TransformationSet transformationSet )
123         throws MojoExecutionException, MojoFailureException
124     {
125 
126         TransformerFactory tf = getTransformerFactory();
127         if ( pResolver != null )
128         {
129             tf.setURIResolver( pResolver );
130         }
131         NameValuePair[] features = transformationSet.getFeatures();
132         if ( features != null )
133         {
134             for ( int i = 0;  i < features.length;  i++ )
135             {
136                 final NameValuePair feature = features[i];
137                 final String name = feature.getName();
138                 if ( name == null  ||  name.length() == 0 )
139                 {
140                     throw new MojoFailureException( "A features name is missing or empty." );
141                 }
142                 final String value = feature.getValue();
143                 if ( value == null )
144                 {
145                     throw new MojoFailureException( "No value specified for feature " + name );
146                 }
147                 setFeature( tf, name, Boolean.valueOf( value ) );
148             }
149         }
150         NameValuePair[] attributes = transformationSet.getAttributes();
151         if ( attributes != null )
152         {
153             for ( int i = 0;  i < attributes.length;  i++ )
154             {
155                 final NameValuePair attribute = attributes[i];
156                 final String name = attribute.getName();
157                 if ( name == null  ||  name.length() == 0 )
158                 {
159                     throw new MojoFailureException( "An attributes name is missing or empty." );
160                 }
161                 final String value = attribute.getValue();
162                 if ( value == null )
163                 {
164                     throw new MojoFailureException( "No value specified for attribute " + name );
165                 }
166                 tf.setAttribute( name, value );
167             }
168         }
169         try
170         {
171             return tf.newTemplates( stylesheet );
172         }
173         catch ( TransformerConfigurationException e )
174         {
175             throw new MojoExecutionException( "Failed to parse stylesheet " + stylesheet + ": " + e.getMessage(), e );
176         }
177     }

179     /**
180      * Creates a new instance of {@link TransformerFactory}.
181      */
182     private TransformerFactory getTransformerFactory( ) throws MojoFailureException, MojoExecutionException
183     {
184         if ( transformerFactory == null )
185         {
186             return TransformerFactory.newInstance();
187         }
188         
189         try
190         {
191             return newTransformerFactory( transformerFactory, Thread.currentThread().getContextClassLoader() );
192         }
193         catch ( NoSuchMethodException exception )
194         {
195             throw new MojoFailureException( "JDK6 required when using transformerFactory parameter" );
196         }
197         catch ( IllegalAccessException exception )
198         {
199             throw new MojoExecutionException( "Cannot instantiate transformer factory", exception );
200         }
201         catch ( InvocationTargetException exception )
202         {
203             throw new MojoExecutionException( "Cannot instantiate transformer factory", exception );
204         }
205     }

207     // public for use by unit test
208     public static TransformerFactory newTransformerFactory( String factoryClassName, ClassLoader classLoader )
209         throws NoSuchMethodException, IllegalAccessException, InvocationTargetException
210     {
211         // use reflection to avoid JAXP 1.4 (and hence JDK6) requirement
212 
213         Class[] methodTypes = new Class[] { String.class, ClassLoader.class };
214         
215         Method method = TransformerFactory.class.getDeclaredMethod( "newInstance", methodTypes );
216         
217         Object[] methodArgs = new Object[] { factoryClassName, classLoader };
218         
219         return (TransformerFactory) method.invoke( null, methodArgs );
220     }

274     private static String getAllExMsgs( Throwable ex, boolean includeExName )
275     {
276         StringBuffer sb = new StringBuffer( ( includeExName ? ex.toString() : ex.getLocalizedMessage() ) );
277         while ( ( ex = ex.getCause() ) != null )
278         {
279             sb.append( "\nCaused by: " + ex.toString() );
280         }
281 
282         return sb.toString();
283     }

285     /**
286      * 
287      * @param files the fileNames or URLs to scan their lastModified timestamp.
288      * @param oldest if true, returns the latest modificationDate of all files,
289      *      otherwise returns the earliest.
290      * @return the older or younger last modification timestamp of all files.
291      */
292     protected long findLastModified( List/*<Object>*/files, boolean oldest )
293     {
294         long timeStamp = ( oldest ? Long.MIN_VALUE : Long.MAX_VALUE );
295         for ( Iterator it = files.iterator(); it.hasNext(); )
296         {
297             Object no = it.next();
298 
299             if ( no != null )
300             {
301                 long fileModifTime;
302                 if ( no instanceof File )
303                 {
304                     fileModifTime = ( (File) no ).lastModified();
305                 }
306                 else // either URL or filePath
307                 {
308                     
309                     String sdep = no.toString();
310 
311                     try
312                     {
313                         URL url = new URL( sdep );
314 
315                         URLConnection uCon = url.openConnection();
316                         uCon.setUseCaches( false );
317 
318                         fileModifTime = uCon.getLastModified();
319 
320                     }
321                     catch ( MalformedURLException e )
322                     {
323                         fileModifTime = new File( sdep ).lastModified();
324 
325                     }
326                     catch ( IOException ex )
327                     {
328                         fileModifTime = ( oldest ? Long.MIN_VALUE : Long.MAX_VALUE );
329                         getLog().warn( "Skipping URL '" + no
330                                            + "' from up-to-date check due to error while opening connection: "
331                                            + getAllExMsgs( ex, true ) );
332                     }
333 
334                 }
335 
336                 getLog().debug( ( oldest ? "Depends " : "Produces " ) + no + ": " + new Date( fileModifTime ) );
337 
338                 if ( ( fileModifTime > timeStamp ) ^ !oldest )
339                 {
340                     timeStamp = fileModifTime;
341                 }
342             } // end if file null.
343         } // end filesloop
344 
345         if ( timeStamp == Long.MIN_VALUE )
346         {
347             // no older file found
348             return Long.MAX_VALUE; // assume re-execution required. 
349         }
350         else if ( timeStamp == Long.MAX_VALUE )
351         {
352             // no younger file found
353             return Long.MIN_VALUE; // assume re-execution required.
354         }
355 
356         return timeStamp;
357     }

359     /**
360      * @return true to indicate results are up-to-date, that is, when the latest 
361      *          from input files is earlier than the younger from the output 
362      *          files (meaning no re-execution required).
363      */
364     protected boolean isUpdToDate( List dependsFiles, List producesFiles )
365     {
366         // The older timeStamp of all input files;
367         long inputTimeStamp = findLastModified( dependsFiles, true );
368 
369         // The younger of all destination files.
370         long destTimeStamp = producesFiles == null
371             ? Long.MIN_VALUE : findLastModified( producesFiles, false ); 
372     
373             getLog().debug( "Depends timeStamp: " + inputTimeStamp + ", produces timestamp: " + destTimeStamp );
374 
375         return inputTimeStamp < destTimeStamp; 
376     }

378     private void transform( Transformer pTransformer, File input, File output, Resolver pResolver )
379         throws MojoExecutionException
380     {
381         File dir = output.getParentFile();
382         dir.mkdirs();
383         getLog().info( "Transforming file: " + input.getPath() );
384         FileOutputStream fos = null;
385         try
386         {
387         	final boolean transformInPlace = output.equals( input );
388         	File tmpOutput = null;
389         	if ( transformInPlace ) {
390         		tmpOutput = File.createTempFile( "xml-maven-plugin", "xml" );
391         		tmpOutput.deleteOnExit();
392                 fos = new FileOutputStream( tmpOutput );
393         	}
394         	else
395         	{
396         		fos = new FileOutputStream( output );
397         	}
398 
399         	final String parentFile = input.getParent() == null
400                 ? null : input.getParentFile().toURI().toURL().toExternalForm();
401             pTransformer.transform( pResolver.resolve( input.toURI().toURL().toExternalForm(),
402                                                        parentFile ), new StreamResult( fos ) );
403             fos.close();
404             fos = null;
405             if ( transformInPlace ) {
406             	FileUtils.copyFile( tmpOutput, output );
407             	/* tmpOutput is a temporary file */
408             	tmpOutput.delete();
409             }
410         }
411         catch ( IOException e )
412         {
413             throw new MojoExecutionException( "Failed to create output file " + output.getPath() + ": "
414                 + e.getMessage(), e );
415         }
416         catch ( TransformerException e )
417         {
418             throw new MojoExecutionException( "Failed to transform input file " + input.getPath() + ": "
419                 + e.getMessage(), e );
420         }
421         finally
422         {
423             if ( fos != null )
424             {
425                 try
426                 {
427                     fos.close();
428                 }
429                 catch ( Throwable t )
430                 {
431                     /* Ignore me */
432                 }
433             }
434         }
435     }

450     private void transform( Resolver pResolver, TransformationSet pTransformationSet )
451         throws MojoExecutionException, MojoFailureException
452     {
453         String[] fileNames = getFileNames( pTransformationSet.getDir(),
454                                            pTransformationSet.getIncludes(),
455                                            getExcludes( pTransformationSet.getExcludes(),
456                                                         pTransformationSet.isSkipDefaultExcludes() ) );
457         if ( fileNames == null || fileNames.length == 0 )
458         {
459             getLog().warn( "No files found for transformation by stylesheet " + pTransformationSet.getStylesheet() );
460             return;
461         }
462 
463         String stylesheetName = pTransformationSet.getStylesheet();
464         if ( stylesheetName == null )
465         {
466             getLog().warn( "No stylesheet configured." );
467             return;
468         }
469 
470         final URL stylesheetUrl = getResource( stylesheetName );
471         Templates template;
472         InputStream stream = null;
473         try
474         {
475         	stream = stylesheetUrl.openStream();
476         	InputSource isource = new InputSource( stream );
477         	isource.setSystemId( stylesheetUrl.toExternalForm() );
478         	template = getTemplate( pResolver, new SAXSource( isource ), pTransformationSet );
479         	stream.close();
480         	stream = null;
481         }
482         catch ( IOException e )
483         {
484         	throw new MojoExecutionException( e.getMessage(), e );
485         }
486         finally
487         {
488         	IOUtil.close( stream );
489         }
490         
491         int filesTransformed = 0;
492         File inputDir = getDir( pTransformationSet.getDir() );
493         File outputDir = getOutputDir( pTransformationSet.getOutputDir() );
494         for ( int i = 0; i < fileNames.length; i++ )
495         {
496             final Transformer t;
497             
498             File input = getFile( inputDir, fileNames[i] );
499             File output = getOutputFile( outputDir, fileNames[i], pTransformationSet.getFileMappers() );
500 
501             // Perform up-to-date-check.
502             boolean needsTransform = forceCreation;
503             if ( !needsTransform )
504             {
505                 List dependsFiles = new ArrayList();
506                 List producesFiles = new ArrayList();
507 
508                 // Depends from pom.xml file for when project configuration changes.
509                 dependsFiles.add( getProject().getFile() );
510                 if ( "file".equals( stylesheetUrl.getProtocol() ) )
511                 {
512                 	dependsFiles.add( new File( stylesheetUrl.getFile() ) );
513                 }
514                 List catalogFiles = new ArrayList();
515                 List catalogUrls = new ArrayList();
516                 setCatalogs( catalogFiles, catalogUrls );
517                 dependsFiles.addAll( catalogFiles );
518                 dependsFiles.add( input );
519                 File[] files = asFiles( getBasedir(), pTransformationSet.getOtherDepends() );
520                 for ( int j = 0;  j < files.length;  j++ )
521                 {
522                     dependsFiles.add( files[j] );
523                 }
524 
525                 producesFiles.add( output );
526                 
527                 needsTransform = !isUpdToDate( dependsFiles, producesFiles );
528             }
529             
530             if ( !needsTransform )
531             {
532                 getLog().debug( "Skipping XSL transformation.  File " + fileNames[i] + " is up-to-date." );
533             }
534             else
535             {
536                 filesTransformed++;
537                 
538                 // Perform transformation.
539                 try
540                 {
541                     t = newTransformer( template, pTransformationSet );
542                     t.setURIResolver( pResolver );
543 
544                     NameValuePair[] parameters = pTransformationSet.getParameters();
545                     if ( parameters != null )
546                     {
547                         for ( int j = 0;  j < parameters.length;  j++  )
548                         {
549                             NameValuePair key = parameters[j];
550                             t.setParameter( key.getName(), key.getValue() );
551                         }
552                     }
553                     
554                     transform( t, input, output, pResolver );
555                     
556                 }
557                 catch ( TransformerConfigurationException e )
558                 {
559                     throw new MojoExecutionException( "Failed to create Transformer: " + e.getMessage(), e );
560                 }
561             }
562         } // end file loop
563 
564         if ( filesTransformed > 0 )
565         {
566             getLog().info( "Transformed " + filesTransformed + " file(s)." );
567         }
568         
569         if ( pTransformationSet.isAddedToClasspath() )
570         {
571             addToClasspath( pTransformationSet.getOutputDir() );
572         }
573     }

575     private Transformer newTransformer( Templates template, TransformationSet pTransformationSet )
576         throws TransformerConfigurationException, MojoExecutionException, MojoFailureException
577     {
578         Transformer t = template.newTransformer();
579         NameValuePair[] properties = pTransformationSet.getOutputProperties();
580         if ( properties != null )
581         {
582             for ( int i = 0;  i < properties.length;  i++ )
583             {
584                 final String name = properties[i].getName();
585                 if ( name == null  ||  "".equals( name ) )
586                 {
587                     throw new MojoFailureException( "Missing or empty output property name" );
588                 }
589                 final String value = properties[i].getValue();
590                 if ( value == null )
591                 {
592                     throw new MojoFailureException( "Missing value for output property " + name );
593                 }
594                 try
595                 {
596                     t.setOutputProperty( name, value );
597                 }
598                 catch ( IllegalArgumentException e )
599                 {
600                     throw new MojoExecutionException( "Unsupported property name or value: "
601                                                       + name + " => "
602                                                       + value
603                                                       + e.getMessage(), e );
604                 }
605             }
606         }
607         return t;
608     }

610     /**
611      * Called by Maven to run the plugin.
612      */
613     public void execute()
614         throws MojoExecutionException, MojoFailureException
615     {
616         if ( transformationSets == null || transformationSets.length == 0 )
617         {
618             throw new MojoFailureException( "No TransformationSets configured." );
619         }
620 
621         Object oldProxySettings = activateProxy();
622         try
623         {
624             Resolver resolver = getResolver( );
625             for ( int i = 0; i < transformationSets.length; i++ )
626             {
627                 TransformationSet transformationSet = transformationSets[i];
628                 resolver.setValidating( transformationSet.isValidating() );
629                 transform( resolver, transformationSet );
630             }
631         }
632         finally
633         {
634             passivateProxy( oldProxySettings );
635         }
636     }
