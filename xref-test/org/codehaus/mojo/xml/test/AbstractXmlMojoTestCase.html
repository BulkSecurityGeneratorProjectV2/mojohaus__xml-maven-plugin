16  package org.codehaus.mojo.xml.test;
17  
18  import java.io.File;
19  import java.io.IOException;
20  import java.lang.reflect.InvocationTargetException;
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  import javax.xml.parsers.DocumentBuilderFactory;
25  import javax.xml.parsers.ParserConfigurationException;
26  
27  import org.apache.maven.model.Build;
28  import org.apache.maven.plugin.logging.Log;
29  import org.apache.maven.plugin.testing.AbstractMojoTestCase;
30  import org.apache.maven.plugin.testing.SilentLog;
31  import org.apache.maven.plugin.testing.stubs.MavenProjectStub;
32  import org.codehaus.mojo.xml.AbstractXmlMojo;
33  import org.codehaus.mojo.xml.TransformMojo;
34  import org.codehaus.plexus.resource.DefaultResourceManager;
35  import org.codehaus.plexus.resource.loader.FileResourceLoader;
36  import org.codehaus.plexus.resource.loader.JarResourceLoader;
37  import org.codehaus.plexus.resource.loader.ResourceLoader;
38  import org.codehaus.plexus.resource.loader.ThreadContextClasspathResourceLoader;
39  import org.codehaus.plexus.resource.loader.URLResourceLoader;
40  import org.w3c.dom.Document;
41  import org.xml.sax.SAXException;
42  
43  public abstract class AbstractXmlMojoTestCase
44      extends AbstractMojoTestCase
45  {
46  
47      protected abstract String getGoal();
48  
49      protected AbstractXmlMojo newMojo( String pDir )
50          throws Exception
51      {
52          File testPom = new File( new File( getBasedir(), pDir ), "pom.xml" );
53          AbstractXmlMojo vm = (AbstractXmlMojo) lookupMojo( getGoal(), testPom );
54          setVariableValueToObject( vm, "basedir", new File( getBasedir(), pDir ) );
55          final Log log = new SilentLog();
56          DefaultResourceManager rm = new DefaultResourceManager();
57          setVariableValueToObject( rm, "logger", log );
58          setVariableValueToObject( vm, "locator", rm );
59          final Map<String, ResourceLoader> resourceLoaders = new HashMap<String, ResourceLoader>();
60          resourceLoaders.put( "file", new FileResourceLoader() );
61          resourceLoaders.put( "jar", new JarResourceLoader() );
62          resourceLoaders.put( "classloader", new ThreadContextClasspathResourceLoader() );
63          URLResourceLoader url = new URLResourceLoader();
64          setVariableValueToObject( url, "logger", log );
65          resourceLoaders.put( "url", url );
66          setVariableValueToObject( rm, "resourceLoaders", resourceLoaders );
67  
68          final Build build = new Build();
69          build.setDirectory( "target" );
70          MavenProjectStub project = new MavenProjectStub()
71          {
72              public Build getBuild()
73              {
74                  return build;
75              }
76          };
77          setVariableValueToObject( vm, "project", project );
78          return vm;
79      }
80  
81      protected void runTest( final String pDir )
82          throws Exception
83      {
84          newMojo( pDir ).execute();
85      }
86  
87      protected Document parse( File pFile )
88          throws SAXException, IOException, ParserConfigurationException
89      {
90          DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
91          dbf.setValidating( false );
92          dbf.setNamespaceAware( true );
93          return dbf.newDocumentBuilder().parse( pFile );
94      }
95  
96      protected boolean java1_6_Aware()
97          throws IllegalAccessException, InvocationTargetException
98      {
99          try
100         {
101             TransformMojo.newTransformerFactory( "net.sf.saxon.TransformerFactoryImpl",
102                                                  Thread.currentThread().getContextClassLoader() );
103             return true;
104         }
105         catch ( NoSuchMethodException e )
106         {
107             return false;
108         }
109     }
110 }
